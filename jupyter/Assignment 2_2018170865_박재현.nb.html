# Confusion Matrix
cm_train <- xtabs(~ train$Survived + fitted_train) # train
cm_test <- xtabs(~ test$Survived + fitted_test) # test

perf_measure <- function(cm) {
  TPR <- cm[2,2]/sum(cm[2,]) # recall
  PRE <- cm[2,2]/sum(cm[,2]) # precision
  TNR <- cm[1,1]/sum(cm[1,]) # True Negative
  ACC <- (cm[1,1]+cm[2,2])/sum(cm)
  BCR <- sqrt(TPR*TNR)
  F1 <- 2*TPR*PRE/(TPR+PRE)
  
  return(c(ACC, BCR, F1))
}

변수 Age의 대한 나의 생각은 노인과 아이가 모두 우선 구출되었을 것이라는 판단이었다. 즉, Age 변수는 별로 유의하지 않을 것이라고 생각했는데, 모델에 의하면 매우 유의한 변수이고, 나이가 한 단위 증가할 수록 -0.04만 큼 예측값이 감소한다.

실제로 아이들이 많이 생존했을지에 대해 분석을 해보고자 한다. Child 변수를 만들어서 Age변수 대신 사용한다. 다른 변수들은 비교, 분석을 위해 그대로 사용한다.
